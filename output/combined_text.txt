# 1. Entropy Calculation
def calculate_entropy(data):
labels = data.iloc[:, -1]
counts = Counter(labels)
probabilities = [count / len(labels) for count in counts.values()]
return -sum(p * np.log2(p) for p in probabilities if p > 0)

# 2. Information Gain Calculation
def calculate_information_gain(data, feature):
total_entropy = calculate_entropy(data)
values = data[feature].unique()

weighted_entropy = 0
for val in values:
subset = data[data[feature] == val]
weighted_entropy += (len(subset) / len(data)) * calculate_entropy(subset)

return total_entropy - weighted_entropy

# 3. The ID3 Algorithm
def id3(data, features, target_col):
labels = data[target_col]

# Base Case 1: All records belong to the same class
if len(labels.unique()) == 1:
return labels.iloc[0]

# Base Case 2: Feature set is empty
if not features:
return labels.mode()[0]

# Select feature with highest Information Gain
gains = {feat: calculate_information_gain(data, feat) for feat in features}
best_feat = max(gains, key=gains.get)

tree = {best_feat: {}}
remaining_features = [f for f in features if f != best_feat]

# Create branches for each value of the best feature
for val in data[best_feat].unique():
subset = data[data[best_feat] == val]

if subset.empty:
tree[best_feat][val] = labels.mode()[0]
else:
tree[best_feat][val] = id3(subset, remaining_features, target_col)

return tree

# 4. Prediction Function
def predict(query, tree, default_label):
for feature in list(query.index):
if feature in tree:
try:
result = tree[feature][query[feature]]
if isinstance(result, dict):
return predict(query, result, default_label)
else:
return result
except:
return default_label
return default_label

# 5. Metrics Calculation
def get_metrics(y_true, y_pred):
accuracy = np.mean(y_true == y_pred)

# For precision, we assume a binary classification (positive/negative)
# If multiclass, this logic needs expansion
classes = np.unique(y_true)
precision_dict = {}

for cls in classes:
tp = np.sum((y_true == cls) & (y_pred == cls))
fp = np.sum((y_true != cls) & (y_pred == cls))
precision_dict[cls] = tp / (tp + fp) if (tp + fp) > 0 else 0

return accuracy, precision_dict

# 3. ID3 CORE
class Node:
def __init__(self, attr=None, lbl=None, maj=None):
self.attr = attr
self.lbl = lbl
self.kids = {}
self.maj = maj

def get_ent(lbls):
cnts = Counter(lbls)
n = len(lbls)
return -sum((c/n) * math.log2(c/n) for c in cnts.values() if c > 0)

def get_ig(data, attr, targ):
p_ent = get_ent(data[targ])
n = len(data)
w_ent = sum((len(s)/n) * get_ent(s[targ]) for _, s in data.groupby(attr))
return p_ent - w_ent

def build_tree(data, attrs, targ, p_maj=None):
lbls = data[targ]
if len(lbls.unique()) == 1: return Node(lbl=lbls.iloc[0])
if data.empty: return Node(lbl=p_maj)
if not attrs: return Node(lbl=lbls.mode()[0])

curr_maj = lbls.mode()[0]
gains = [get_ig(data, a, targ) for a in attrs]
best = attrs[np.argmax(gains)]

root = Node(attr=best, maj=curr_maj)
rem = [a for a in attrs if a != best]

for val in data[best].unique():
sub = data[data[best] == val]
root.kids[val] = build_tree(sub, rem, targ, curr_maj)
return root

# 4. PREDICT
def pred(node, inst):
if node.lbl is not None: return node.lbl
val = inst[node.attr]
if val not in node.kids: return node.maj
return pred(node.kids[val], inst)
